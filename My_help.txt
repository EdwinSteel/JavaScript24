arrayName.length

// Добавить элемент в массив (по умолчанию всегда добвляется в конец массива)

arrayName[0] = 50 -- добавит значение "50" в позицию под индексом "0"
arrayName[0] = 60 -- перезапишет значение в позиции  под индексом "0"

// добавить в конец массива
arrayName[arrayName.length] = 77;
arrayName[arrayName.length] = 88;
console.log(arrayName);
console.log(arrayName.length);
-----------------------------------------------

// методы для работы с массивами
arrayName.push(element-1) -- добавит значение "element-1" в конец массива и возвращает длину массива

arrayName.push(element-1, element-2, element-3) -- добавит значения "element-1", "element-2", "element-3" в конец массива

arrayName.pop() -- удаляет последний элемент из массива и возвращает удаленный элемент массива

arrayName.unshift(element-1) -- добавляет "element-1" в начало массива
и возвращает новую длину массива

arrayName.shift() -- удаляет элемент в начале массива

ИТОГ:
push(element-1) -- добавит значения в конец массива
pop() -- удаляет последний элемент из массива и возвращает удаленный элемент массива
arrayName.unshift(element-1) -- добавляет в начало массива
и возвращает новую длину массива
arrayName.shift() -- удаляет элемент в начале массива
